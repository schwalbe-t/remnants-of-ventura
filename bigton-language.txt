# BIGTON Processor Language:

# Basic Language:
# - Global variables (number limited by processor type)
# - Arithmetic
# - Loops ('tick', 'loop', 'while', 'continue', 'break')
# - Number Literals
# - String Literals
# - Unnamed Tuples
# - Limit on number of total executed instructions every tick
#   (depending on processor type)

# 'var' is block-scoped (or global)
var a = 0
var b = 1
var c = 0
var n = 100

while n > 0 {
    c = a + b
    a = b
    b = c
    n = n - 1
}

# stops executing below and instead runs every game tick
tick {
    say(n)
}


# Language Extensions (depending on processor)
# - Floating point module
var test = 3.4
say(sqrt(test))

# - RAM module (size depends on processor)
var nums = 100
*(nums+0) = 6
*(nums+1) = 7
*(nums+2) = 8
*(nums+3) = 9

# - Custom instructions / functions
#     - 'fun'
#     - 'return'
fun add(a, b) {
    return a + b
}

# - Custom types
#     - Literals
#     - Member accesses
var cat = { name: "Bob", age: 5 }
say(cat.name)
